# Problem Copy Block — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Deter students from copying problem descriptions to external tools via `user-select: none` CSS + warning banner + beforeunload dialog, controlled per-problem.

**Architecture:** One new boolean field on Problem model (`block_copy`, default `False`). Template condition in `problem.html` wraps the description in `user-select: none` and shows a warning banner for non-staff, non-editor users. A `beforeunload` event listener warns on navigation away. No JS event blocking on copy, no DB violation recording.

**Tech Stack:** Django, PostgreSQL migration, vanilla JS, Django templates (django_jinja)

## Global Constraints

- Copy buttons on `<pre><code>` blocks must remain functional
- Staff and problem editors are exempt from all blocking
- `block_copy` defaults to `False` (disabled)
- No DB violation recording

- [ ] **Step 2: Add beforeunload JS (below the existing PDF print script)**

After the existing `{% block content_js_media %}` closing tag (line 75), add:

```html
{% block content_js_media %}
    {# Keep existing PDF print script exactly as-is #}
    {% if not comment_lock %}
        {% include "comments/media-js.html" %}
    {% endif %}
    <script type="text/javascript">
        $(function() {
            $('#pdf_button').click(async function(e) {
                e.preventDefault();
                if (!$('#raw_problem').attr("src"))
                    $('#raw_problem').attr("src", "{{problem.code}}/raw");
                while(!$('.math-loaded', frames['raw_problem'].document).length) {
                    await new Promise(r => setTimeout(r, 200));
                }
                frames['raw_problem'].print();
            });
        });
    </script>

    {% if problem.block_copy and not request.user.is_staff and not can_edit_problem %}
    <script>
        (function() {
            var blockCopy = true;
            window.addEventListener('beforeunload', function(e) {
                if (!blockCopy) return;
                e.preventDefault();
                e.returnValue = '';
            });
            // Suppress beforeunload on intentional navigation (clicks on links/buttons)
            document.addEventListener('click', function(e) {
                var a = e.target.closest('a');
                if (a && a.href && !a.href.startsWith('javascript:')) {
                    blockCopy = false;
                }
            });
            // Also suppress on form submission
            document.addEventListener('submit', function() {
                blockCopy = false;
            }, true);
        })();
    </script>
    {% endif %}
{% endblock %}
```

Note: the existing `content_js_media` block (lines 57-75) will be extended rather than replaced. The `{% block content_js_media %}` in the template already wraps the PDF script — we add the beforeunload script inside that same block, after the PDF script.

from django.test import TestCase
from judge.models import Problem, ProblemGroup, ProblemType


class ProblemBlockCopyTest(TestCase):
    @classmethod
    def setUpTestData(cls):
        cls.group = ProblemGroup.objects.create(name='Test Group', full_name='Test Group')
        cls.problem_type = ProblemType.objects.create(name='Test Type')

    def test_block_copy_defaults_to_false(self):
        problem = Problem.objects.create(
            code='testblock',
            name='Test Block Copy',
            time_limit=1,
            memory_limit=65536,
            group=self.group,
        )
        problem.types.add(self.problem_type)
        self.assertFalse(problem.block_copy)

    def test_block_copy_can_be_set_true(self):
        problem = Problem.objects.create(
            code='testblock2',
            name='Test Block Copy 2',
            time_limit=1,
            memory_limit=65536,
            group=self.group,
            block_copy=True,
        )
        problem.types.add(self.problem_type)
        self.assertTrue(problem.block_copy)

```python
from django.urls import reverse
from judge.models.tests.util import create_user


class ProblemBlockCopyViewTest(TestCase):
    @classmethod
    def setUpTestData(cls):
        cls.group = ProblemGroup.objects.create(name='Test Group', full_name='Test Group')
        cls.problem_type = ProblemType.objects.create(name='Test Type')
        cls.problem = Problem.objects.create(
            code='viewtest',
            name='View Test',
            time_limit=1,
            memory_limit=65536,
            description='# Hello\n\nThis is a problem description.',
            group=cls.group,
            block_copy=True,
        )
        cls.problem.types.add(cls.problem_type)
        cls.student = create_user(username='student', password='password')
        cls.staff_user = create_user(username='staff', password='password',
                                      user_permissions=('change_problem',))

    def test_blocked_user_sees_banner(self):
        self.client.login(username='student', password='password')
        response = self.client.get(reverse('problem_detail', args=('viewtest',)))
        self.assertEqual(response.status_code, 200)
        self.assertContains(response, 'block-copy-banner')
        self.assertContains(response, 'user-select: none')

    def test_staff_user_does_not_see_banner(self):
        self.client.login(username='staff', password='password')
        response = self.client.get(reverse('problem_detail', args=('viewtest',)))
        self.assertEqual(response.status_code, 200)
        self.assertNotContains(response, 'block-copy-banner')
        self.assertNotContains(response, 'user-select: none')

    def test_block_copy_false_no_banner(self):
        self.problem.block_copy = False
        self.problem.save()
        self.client.login(username='student', password='password')
        response = self.client.get(reverse('problem_detail', args=('viewtest',)))
        self.assertEqual(response.status_code, 200)
        self.assertNotContains(response, 'block-copy-banner')
```
